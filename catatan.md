// Array static
``` java
public static void main(String[] args) {
	Integer a[] = {3,9,1,0};
	System.out.print(a[0]);
	System.out.print(a[1]);
	System.out.print(a[2]);
	System.out.print(a[3]);

	/*jika di tambah lebih dari index*/
	System.out.print(a[4]);
	a[3]=11;
	System.out.print(a[3]);
}
```
